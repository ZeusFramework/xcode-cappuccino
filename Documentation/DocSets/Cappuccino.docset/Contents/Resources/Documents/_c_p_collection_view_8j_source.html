00359 - (void)tile
00360 {
00361     var width = CGRectGetWidth([self bounds]);
00362         
00363     if (![_content count] || width == _tileWidth)
00364         return;
00365         
00366     // We try to fit as many views per row as possible.  Any remaining space is then 
00367     // either proportioned out to the views (if their minSize != maxSize) or used as
00368     // margin
00369     var itemSize = CGSizeMakeCopy(_minItemSize);
00370     
00371     _numberOfColumns = MAX(1.0, FLOOR(width / itemSize.width));
00372     
00373     if (_maxNumberOfColumns > 0)
00374         _numberOfColumns = MIN(_maxNumberOfColumns, _numberOfColumns);
00375             
00376     var remaining = width - _numberOfColumns * itemSize.width,
00377         itemsNeedSizeUpdate = NO;
00378         
00379     if (remaining > 0 && itemSize.width < _maxItemSize.width)
00380         itemSize.width = MIN(_maxItemSize.width, itemSize.width + FLOOR(remaining / _numberOfColumns));
00381     
00382     // When we ONE column and a non-integral width, the FLOORing above can cause the item width to be smaller than the total width.
00383     if (_maxNumberOfColumns == 1 && itemSize.width < _maxItemSize.width && itemSize.width < width)
00384         itemSize.width = MIN(_maxItemSize.width, width);
00385     
00386     if (!CGSizeEqualToSize(_itemSize, itemSize))
00387     {
00388         _itemSize = itemSize;
00389         itemsNeedSizeUpdate = YES;
00390     }
00391     
00392     var index = 0,
00393         count = _items.length;
00394     
00395     if (_maxNumberOfColumns > 0 && _maxNumberOfRows > 0)
00396         count = MIN(count, _maxNumberOfColumns * _maxNumberOfRows);
00397     
00398     _numberOfRows = CEIL(count / _numberOfColumns);
00399 
00400     _horizontalMargin = FLOOR((width - _numberOfColumns * itemSize.width) / (_numberOfColumns + 1));
00401         
00402     var x = _horizontalMargin,
00403         y = -itemSize.height;
00404     
00405     for (; index < count; ++index)
00406     {
00407         if (index % _numberOfColumns == 0)
00408         {
00409             x = _horizontalMargin;
00410             y += _verticalMargin + itemSize.height;
00411         }
00412         
00413         var view = [_items[index] view];
00414         
00415         [view setFrameOrigin:CGPointMake(x, y)];
00416         
00417         if (itemsNeedSizeUpdate)
00418             [view setFrameSize:_itemSize];
00419             
00420         x += itemSize.width + _horizontalMargin;
00421     }
00422     
00423     _tileWidth = width;
00424     [self setFrameSize:CGSizeMake(width, y + itemSize.height + _verticalMargin)];
00425     _tileWidth = -1.0;
